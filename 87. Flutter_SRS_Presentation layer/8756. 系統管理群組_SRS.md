class SystemManagementStateManager {
  SystemManagementState _currentState = SystemManagementState.idle;
  final StreamController<SystemManagementState> _stateController = StreamController.broadcast();
  final StreamController<Map<String, dynamic>> _dataController = StreamController.broadcast();
  
  Stream<SystemManagementState> get stateStream => _stateController.stream;
  Stream<Map<String, dynamic>> get dataStream => _dataController.stream;
  SystemManagementState get currentState => _currentState;
  
  void updateState(SystemManagementState newState, {Map<String, dynamic>? data}) {
    if (_currentState != newState) {
      _currentState = newState;
      _stateController.add(newState);
      _handleStateChange(newState);
    }
    
    if (data != null) {
      _dataController.add(data);
    }
  }
  
  void _handleStateChange(SystemManagementState state) {
    switch (state) {
      case SystemManagementState.monitoring:
        _startMonitoringTimer();
        break;
      case SystemManagementState.settingsSaving:
        _showSavingIndicator();
        break;
      case SystemManagementState.error:
        _showErrorRecoveryOptions();
        break;
      case SystemManagementState.maintenance:
        _showMaintenanceNotice();
        break;
      default:
        break;
    }
  }
  
  void _startMonitoringTimer() {
    Timer.periodic(Duration(seconds: 30), (timer) {
      if (_currentState != SystemManagementState.monitoring) {
        timer.cancel();
        return;
      }
      
      _refreshMonitoringData();
    });
  }
  
  Future<void> _refreshMonitoringData() async {
    try {
      final monitoringData = await SystemMonitoringService.getLatestMetrics();
      _dataController.add({
        'type': 'monitoring_update',
        'data': monitoringData,
        'timestamp': DateTime.now()
      });
    } catch (e) {
      updateState(SystemManagementState.error, data: {
        'error': e.toString(),
        'source': 'monitoring_refresh'
      });
    }
  }
}
