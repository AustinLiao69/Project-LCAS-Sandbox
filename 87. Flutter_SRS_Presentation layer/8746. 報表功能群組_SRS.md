class ReportStateManager {
  ReportState _currentState = ReportState.idle;
  final StreamController<ReportState> _stateController = StreamController.broadcast();
  final StreamController<double> _progressController = StreamController.broadcast();
  
  Stream<ReportState> get stateStream => _stateController.stream;
  Stream<double> get progressStream => _progressController.stream;
  ReportState get currentState => _currentState;
  
  void updateState(ReportState newState, {double? progress}) {
    if (_currentState != newState) {
      _currentState = newState;
      _stateController.add(newState);
      _handleStateChange(newState);
    }
    
    if (progress != null) {
      _progressController.add(progress);
    }
  }
  
  void _handleStateChange(ReportState state) {
    switch (state) {
      case ReportState.generating:
        _startProgressTracking();
        break;
      case ReportState.ready:
        _showCompletionNotification();
        break;
      case ReportState.error:
        _showErrorDialog();
        break;
      case ReportState.expired:
        _showExpiryWarning();
        break;
      default:
        break;
    }
  }
  
  void _startProgressTracking() {
    Timer.periodic(Duration(seconds: 1), (timer) {
      if (_currentState != ReportState.generating) {
        timer.cancel();
        return;
      }
      
      // æ¨¡æ“¬é€²åº¦æ›´æ–°
      double currentProgress = _progressController.hasListener ? 
        _progressController.sink as double : 0.0;
      
      if (currentProgress < 100) {
        _progressController.add(currentProgress + 5);
      }
    });
  }
}
